class Btn extends React.Component {
   events = {};
   constructor(props) {
      super(props);

      this.state = {
         id: props.id,
         text: props.children,
      };
   }
   render() {
      return (
         <button type='button' id={this.state.id}>
            {this.state.text}
         </button>
      );
   }
}
